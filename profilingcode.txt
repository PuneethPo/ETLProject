def profile_datafram_v2(df: pd.DataFrame, sample_size=3):
    profile_data = []
    for col in df.columns:
        col_data = df[col]
        total_count = len(col_data)
        non_null_data = col_data.dropna()
        non_null_count = non_null_data.shape[0]
        null_count = total_count - non_null_count
        null_percentage = (null_count / total_count) * 100 if total_count else 0
        non_null_percentage = 100 - null_percentage
        unique_values = non_null_data.nunique()
        distinctness = unique_values / non_null_count if non_null_count else 0
        uniqueness = unique_values / total_count if total_count else 0

        # String length stats (if applicable)
        if col_data.dtype == object or pd.api.types.is_string_dtype(col_data):
            lengths = non_null_data.astype(str).map(len)
            min_length = lengths.min()
            max_length = lengths.max()
            avg_length = lengths.mean()
        else:
            min_length = max_length = avg_length = None

        # Min, Max value
        min_value = non_null_data.min() if pd.api.types.is_numeric_dtype(col_data) else None
        max_value = non_null_data.max() if pd.api.types.is_numeric_dtype(col_data) else None

        # Outlier detection using IQR
        if pd.api.types.is_numeric_dtype(col_data):
            q1 = non_null_data.quantile(0.25)
            q3 = non_null_data.quantile(0.75)
            iqr = q3 - q1
            lower_bound = q1 - 1.5 * iqr
            upper_bound = q3 + 1.5 * iqr
            outliers = non_null_data[(non_null_data < lower_bound) | (non_null_data > upper_bound)]
            outlier_count = len(outliers)
            outlier_reasons = outliers.tolist()[:5]
        else:
            outlier_count = 0
            outlier_reasons = []

        # Sample values (non-null, unique preferred)
        sample_values = non_null_data.drop_duplicates().sample(min(sample_size, len(non_null_data.drop_duplicates())), random_state=42).tolist()

        profile_data.append({
            "column": col,
            "null_percentage": round(null_percentage, 2),
            "non_null_percentage": round(non_null_percentage, 2),
            "null_count": null_count,
            "non_null_count": non_null_count,
            "missing_count": null_count,
            "missing_percentage": round(null_percentage, 2),
            "unique_values": unique_values,
            "distinctness": round(distinctness, 4),
            "uniqueness": round(uniqueness, 4),
            "min_length": min_length,
            "max_length": max_length,
            "avg_length": round(avg_length, 2) if avg_length is not None else None,
            "min_value": min_value,
            "max_value": max_value,
            "actual_count": total_count,
            "outlier_count": outlier_count,
            "outlier_reasons": outlier_reasons,
            "sample_values": sample_values
        })

    return pd.DataFrame(profile_data)

def get_llm_sample(df, value_col, sample_size=100):
    """
    Get a representative sample of values from a DataFrame column for LLM processing
    Uses stratified sampling for categorical data, random for others
    """
    # First try to get distinct values if there aren't too many
    distinct_vals = df.select(value_col).distinct()
    distinct_count = distinct_vals.count()
    
    if distinct_count <= sample_size:
        return distinct_vals
    
    # For columns with many distinct values, use random sampling
    fraction = min(1.0, sample_size / df.count())
    return df.select(value_col).sample(withReplacement=False, fraction=fraction, seed=42).limit(sample_size)



def robust_sample_dataframe(df: DataFrame, n_rows: int = 50) -> DataFrame:
    # Step 1: Get one non-null row per column (if possible)
    non_null_samples = []
    
    for col_name in df.columns:
        non_null_row = df.filter(df[col_name].isNotNull()).limit(10)
        non_null_samples.append(non_null_row)
    
    # Step 2: Union all non-null examples (to ensure coverage)
    guaranteed_sample = reduce(DataFrame.unionByName, non_null_samples)

    # Step 3: Add some random samples for variety
    random_sample = df.sample(fraction=0.01).limit(n_rows)

    # Step 4: Combine and deduplicate (just in case)
    final_sample = guaranteed_sample.unionByName(random_sample).dropDuplicates()

    return final_sample.limit(n_rows)

def get_column_samples(df, limit=100):
    """
    Systematic/random sample up to `limit` non-null values from each column.
    Uses distributed sampling for efficiency with large datasets.
    """
    samples = {}
    for col_name in df.columns:
        base = df.select(col_name).dropna()
        total_count = base.count()
        
        if total_count <= limit:
            vals = base.rdd.flatMap(lambda x: x).collect()
        else:
            # Systematic sample using distributed index
            df_with_idx = base.withColumn("idx", monotonically_increasing_id())
            stride = max(1, total_count // limit)
            idxs = [i * stride for i in range(limit)]
            sampled = df_with_idx.where(col("idx").isin(idxs)).drop("idx")
            vals = sampled.rdd.flatMap(lambda x: x).collect()
            
            # Fallback to random if needed
            if len(vals) < limit:
                needed = limit - len(vals)
                rest = base.exceptAll(sampled).sample(
                    withReplacement=False, 
                    fraction=min(1.0, needed/total_count), 
                    seed=42
                ).limit(needed)
                vals += rest.rdd.flatMap(lambda x: x).collect()
                
        samples[col_name] = vals[:limit]
    return samples

def get_profiling_table_name(src_table, project_id):
    # Remove any double-prefix
    base = src_table
    # Remove 'Profiling_' prefix if present
    if base.lower().startswith('profiling_'):
        base = base[len('Profiling_'):]
    # Remove leading project_id if present
    if base.startswith(f"{project_id}_"):
        base = base[len(f"{project_id}_"):]
    # Now apply the prefix once
    return f"Profiling_{project_id}_{base}"

def analyze_single_column(df, value_col, desc_col, type_col, table_name):
    """Profile values under a single column with project context"""
    value_series = df[value_col].dropna().astype(str)
    descriptions = dict(zip(df[value_col].astype(str), df[desc_col].astype(str)))
    datatypes = dict(zip(df[value_col].astype(str), df[type_col].astype(str)))
    
    results = []
    unique_vals = value_series.nunique()
    total_count = len(value_series)
    null_count = df[value_col].isna().sum()
    
    for value in value_series.unique():
        count = (value_series == value).sum()
        description = descriptions.get(value, "No description")
        datatype = datatypes.get(value, "unknown")
        
        results.append((
            project_id if project_id else "",  # project_id
            table_name,                       # table_name
            value,                            # column_name
            datatype,                         # data_type
            description,                      # description
            float((null_count / total_count) * 100) if total_count else 0.0,  # null_percentage
            unique_vals,                      # unique_values
            min([len(str(value))]),           # min_length
            max([len(str(value))]),           # max_length
            100.0 if total_count else 0.0,    # non_null_percentage
            100.0 if total_count else 0.0,    # uniqueness
            100.0 if total_count else 0.0,    # distinctness
            0,                                # outlier_count
            "",                               # outlier_reasons
            null_count,                       # null_count
            0,                                # missing_count
            0.0,                              # missing_percentage
            "",                               # min_value
            "",                               # max_value
            float(len(value)),                # avg_length
            total_count,                      # actual_count
            total_count - null_count           # non_null_count
        ))
    return results

def detect_outliers(df, results):
    """Basic outlier detection for numeric columns"""
    numeric_cols = [f.name for f in df.schema if isinstance(f.dataType, (IntegerType, FloatType))]
    
    for i, col_name in enumerate(df.columns):
        if col_name in numeric_cols:
            col_stats = df.select(
                mean(col(col_name)).alias('mean'),
                stddev(col(col_name)).alias('std')
            ).collect()[0]
            
            if col_stats['std'] is not None:
                outliers = df.filter(
                    (abs(col(col_name) - col_stats['mean']) > (3 * col_stats['std']))
                ).count()
                
                if outliers > 0:
                    results[i] = list(results[i])
                    results[i][12] = outliers  # outlier_count
                    results[i][13] = "Values beyond 3 standard deviations"  # outlier_reasons
    
    return results

    """Store profiling results with project context"""

        StructField("data_type", StringType(), True),
        StructField("description", StringType(), True),
        StructField("null_percentage", FloatType(), True),
        StructField("unique_values", IntegerType(), True),
        StructField("min_length", IntegerType(), True),
        StructField("max_length", IntegerType(), True),
        StructField("non_null_percentage", FloatType(), True),
        StructField("uniqueness", FloatType(), True),
        StructField("distinctness", FloatType(), True),
        StructField("outlier_count", IntegerType(), True),

        StructField("null_count", IntegerType(), True),
        StructField("missing_count", IntegerType(), True),
        StructField("missing_percentage", FloatType(), True),
        StructField("min_value", StringType(), True),
        StructField("max_value", StringType(), True),
        StructField("avg_length", FloatType(), True),
        StructField("actual_count", IntegerType(), True),
        StructField("non_null_count", IntegerType(), True)
    ])

    # Convert all numpy types in results to native Python types using `to_py()`
    clean_rows = [Row(*[to_py(val) for val in row]) for row in results]

    df_result = spark.createDataFrame(clean_rows, schema=schema)

    df_result.write.format("jdbc") \
        .option("url", jdbc_url) \
        .option("dbtable", table_name) \
        .mode("overwrite") \
        .save()

    logging.info(f"Stored profiling results to {table_name}")




def to_py(obj):
    """Ensure values are native Python types for Spark"""
    if isinstance(obj, (np.integer, np.int64, np.int32)):
        return int(obj)
    elif isinstance(obj, (np.floating, np.float64, np.float32)):
        return float(obj)
    else:
        return obj

Sample columns: {json.dumps(column_names)}  
Sample rows (up to 20): {json.dumps(data_rows, indent=2)}  

Respond with only 'true' (if this is a data dictionary table) or 'false' (if it is a regular data table).  
"""  
    result = call_openai(prompt)  
    # Accept only 'true' (case-insensitive, stripped) as True  
    return result.strip().lower().startswith("true")



def build_column_role_prompt(df, n=10):  
    sample_rows = df.limit(n).toPandas().to_dict(orient='records')  
    prompt = (  
        "Given the following table columns and sample data rows, determine which COLUMN NAMES correspond to:\n"  
        "1. The column name of a downstream table (the column that contains field/column names from a data dictionary)\n"  
        "2. The data type of the target columns\n"  
        "3. The human-readable description of the columns\n"
        "Return your answer as a JSON object with keys 'column_name', 'data_type', 'description' mapping to the detected column names.\n"  
        "If uncertain, return null for that field.\n\n"  
        f"COLUMNS: {list(df.columns)}\n"  
        f"SAMPLE ROWS:\n{json.dumps(sample_rows, indent=2)}\n"  
        "Answer (JSON only):"  
    )  
    return prompt  


def detect_datadict_columns_with_llm(df):  
    prompt = build_column_role_prompt(df)  
    result_raw = call_openai_v2(prompt).strip()  
    # Clean code block if needed  
    if result_raw.startswith("```json"):  
        result_raw = result_raw[len("```json"):].strip()  
    if result_raw.endswith("```"):  
        result_raw = result_raw[:-3].strip()  
    result = json.loads(result_raw)  
    return (result.get('column_name'), result.get('data_type'), result.get('description'))

def do_profiling_for_data_dict_formatted_target(df, proj_id, target_table_name, llm_detect_func, output_tablename):  
    """  
    Generate profiling results from a data dictionary-like dataframe with arbitrary column names.   
    Args:  
      df: pandas.DataFrame with columns for column_name, data_type, and description (but names arbitrary)  
      proj_id: project ID constant  
      target_table_name: table name constant  
      llm_detect_func: function that takes df and returns (column_name_col, data_type_col, description_col)   
    Returns:  
      List of profiling dicts ready for DataFrame construction.  
    """  
    column_name_col, data_type_col, description_col = llm_detect_func(df)  
    if not all([column_name_col, data_type_col, description_col]):  
        raise ValueError(f"Could not auto-detect column_name, data_type, and description columns. Detected: {(column_name_col, data_type_col, description_col)}.")  
  
    profiling_results = []
    df = df.toPandas()  
    for _, row in df.iterrows():  
        result = {  
            "project_id": proj_id,  
            "table_name": target_table_name,  
            "column_name": str(row[column_name_col]).strip(),  
            "data_type": str(row[data_type_col]).strip(),  
            "description": str(row[description_col]).strip(),  
            # Defaulted profiling/stat fields  
            "null_percentage": 0.0,  
            "unique_values": 0,  
            "min_length": 0,  
            "max_length": 0,  
            "non_null_percentage": 0.0,  
            "uniqueness": 0.0,  
            "distinctness": 0.0,  
            "outlier_count": 0,  
            "outlier_reasons": "",  
            "null_count": 0,  
            "missing_count": 0,  
            "missing_percentage": 0.0,  
            "min_value": "",  
            "max_value": "",  
            "avg_length": 0.0,  
            "actual_count": 0,  
            "non_null_count": 0  
        }  
        profiling_results.append(result)
    save_profiling_results_table(profiling_results, proj_id, target_table_name, output_tablename)
    display(profiling_results)
    return profiling_results

def call_openai_v2(prompt):
    print(prompt)
    """Call Azure OpenAI API for column descriptions"""
    headers = {"Content-Type": "application/json", "api-key": api_key}
    payload = {
        "messages": [
            {"role": "system", "content": "You are a helpful data profiling assistant."},
            {"role": "user", "content": prompt}
        ],
        "max_tokens": 4000,
        "temperature": 0.0
    }
    url = f"{openai_endpoint}/openai/deployments/{openai_deployment}/chat/completions?api-version={api_version}"
    response = requests.post(url, headers=headers, json=payload, timeout=60)
    response.raise_for_status()
    # print(response.json())
    return response.json()["choices"][0]["message"]["content"]

            profile_table_with_gpt(target_df, target_table)
        #profile_target_staging_column(target_df)

        # Profile source tables in parallel
        source_tables = [s.strip() for s in source_table.split(",") if s.strip()]
        
        if not source_tables:
            print("No valid source tables provided")
            return

        print(f"\n=== PROFILING {len(source_tables)} SOURCE TABLES ===")
        
        def profile_source_table(src_table):
            """Wrapper function for parallel profiling"""
            # try:
            print(f"Processing source table: {src_table}")
            src_df = spark.read.format("jdbc") \
                .option("url", jdbc_url) \
                .option("dbtable", src_table) \
                .load()
            
            table_name = src_table
            profff = profile_table_with_gpt(src_df, table_name)
            display(profff)

            
            return True
            # except Exception as e:
            #     print(f"Error profiling {src_table}: {str(e)}")
            #     return False

        # Execute in parallel
        with ThreadPoolExecutor(max_workers=4) as executor:
            futures = [executor.submit(profile_source_table, t) for t in source_tables]
            results = [f.result() for f in as_completed(futures)]
            
        success_count = sum(results)
        print(f"\nCompleted profiling: {success_count}/{len(source_tables)} sources successfully")

        # ===== Workflow Tracking =====
        if success_count > 0:
            print("\nUpdating workflow tracker...")
            workflow_schema = StructType([
                StructField("source_table", StringType(), True),
                StructField("target_table", StringType(), True),
                StructField("profiling_source_table", StringType(), True),
                StructField("profiling_target_table", StringType(), True),
                StructField("mapping_table", StringType(), True),
                StructField("project_id", StringType(), True),
                StructField("status", StringType(), True),
                StructField("created_at", TimestampType(), True)
            ])

            workflow_data = []
            for src_table, success in zip(source_tables, results):
                if success:
                    workflow_data.append(Row(
                                source_table=src_table,
                                target_table=target_table,
                                # profiling_source_table=f"Profiling_{project_id}_{src_table}",
                                # profiling_target_table=f"Profiling_{project_id}_{target_table}",
                                profiling_source_table=get_profiling_table_name(src_table, project_id),
                                profiling_target_table=get_profiling_table_name(target_table, project_id),

                                mapping_table=f"mapping_{project_id}_{src_table}_to_{target_table}",
                                project_id=project_id,
                                status="profiling_completed",
                                created_at=datetime.now()
                            ))

            if workflow_data:
                spark.createDataFrame(workflow_data, schema=workflow_schema) \
                    .write.format("jdbc") \
                    .option("url", jdbc_url) \
                    .option("dbtable", "workflow_tracker") \
                    .mode("append") \
                    .save()
                
                print(f"Updated workflow tracker with {len(workflow_data)} entries")
        
        print("\n=== PROFILING COMPLETED ===")
        
    except Exception as e:
        logging.error(f"Fatal error in main execution: {str(e)}", exc_info=True)
        raise

if __name__ == "__main__":
    main()
