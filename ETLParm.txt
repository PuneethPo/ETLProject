import json
import re

def extract_first_json_array(text):
    """
    Extract the first JSON array from the text using regex.
    Returns the parsed JSON array or raises an error if not found or invalid.
    """
    json_array_match = re.search(r'\[\s*{.*?}\s*]', text, re.DOTALL)
    if not json_array_match:
        raise ValueError("No JSON array found in the GPT response.")
    
    json_str = json_array_match.group()
    return json.loads(json_str)
