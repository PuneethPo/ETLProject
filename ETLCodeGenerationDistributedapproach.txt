def generate_validation_prompt_blocks(df, source_table_names):
    prompt_blocks = []
    issues = []
 
    for source_entity_name in source_table_names:
        filtered_df = df[df['SOURCE_TABLE'].str.contains(source_entity_name, case=False, na=False)]
 
        if filtered_df.empty:
            continue
 
        validation_lines = []
        for idx, row in filtered_df.iterrows():
            try:
                source_col = row.get('SOURCE_COLUMN', None)
                source_dtype = row.get('SOURCE_COLUMN_DATATYPE', None)
                source_mandatory = row.get(' SOURCE_COLUMN_MANDATORY_FIELD', None)
                source_sample = row.get('SOURCE_COLUMN_SAMPLE_VALUE', None)
                target_col = row.get('TARGET_COLUMN', None)
                target_dtype = row.get('TARGET_COLUMN_DATATYPE', None)
                target_mandatory = row.get(' TARGET_COLUMN_MANDATORY_FIELD', None)
                target_sample = row.get('TARGET_COLUMN_SAMPLE_VALUE', None)
                transformation = row.get('TRANSFORMATION', None)
 
                if all(val is not None and pd.notna(val) for val in [
                    source_col, source_dtype, source_mandatory, source_sample,
                    target_col, target_dtype, target_sample, transformation
                ]):
                    validation_lines.append(
                        f"validate source column: '{source_col}' "
                        f"(datatype: '{source_dtype}', mandatory: '{str(source_mandatory).strip()}', "
                        f"sample: '{source_sample}') with rule: '{transformation}'\n"
                        f"validate target column: '{target_col}' "
                        f"(datatype: '{target_dtype}', mandatory: '{str(target_mandatory).strip()}', "
                        f"sample: '{target_sample}') with rule: '{transformation}'\n"
                    )
                else:
                    issues.append(
                        f"Issue in row {idx} of table '{source_entity_name}': "
                        f"Missing or null values in columns: "
                        f"{[col for col, val in [('SOURCE_COLUMN', source_col), ('SOURCE_COLUMN_DATATYPE', source_dtype), (' SOURCE_COLUMN_MANDATORY_FIELD', source_mandatory), ('SOURCE_COLUMN_SAMPLE_VALUE', source_sample), ('TARGET_COLUMN', target_col), ('TARGET_COLUMN_DATATYPE', target_dtype), (' TARGET_COLUMN_MANDATORY_FIELD', target_mandatory), ('TARGET_COLUMN_SAMPLE_VALUE', target_sample), ('TRANSFORMATION', transformation)] if val is None or pd.isna(val)]}"
                    )
            except Exception as e:
                issues.append(f"Exception in row {idx} of table '{source_entity_name}': {str(e)}")
 
        if validation_lines:
            prompt_block = (
                f"Source Table Name: '{source_entity_name}'\n"
                f"Validation Rules To Apply On Columns:\n{''.join(validation_lines)}"
            )
            prompt_blocks.append(prompt_block)
 
    if issues:
        print("Validation prompt block issues:")
        for issue in issues:
            print(issue)
 
    return prompt_blocks

import json
import re

def extract_first_json_array(text):
    """
    Extract the first JSON array from the text using regex.
    Returns the parsed JSON array or raises an error if not found or invalid.
    """
    json_array_match = re.search(r'\[\s*{.*?}\s*]', text, re.DOTALL)
    if not json_array_match:
        raise ValueError("No JSON array found in the GPT response.")
    
    json_str = json_array_match.group()
    return json.loads(json_str)
