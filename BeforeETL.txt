def log_and_raise(e, msg: str):
    """
    Log the error and raise with informative message.
    """
    logging.error(f"{msg}\n{traceback.format_exc()}")
    raise Exception(f"{msg}: {str(e)}")

def format_date_str(col_name, input_formats=None):
    """
    Accepts a column (string or date) and returns formatted YYYY-MM-DD string.
    If value is null or unparsable, returns blank (" ").
    """
    input_formats = input_formats or ["M/d/yyyy", "yyyy-MM-dd", "MM/dd/yyyy", "yyyy/MM/dd"]
    parsed_date = F.coalesce(
        *[F.to_date(F.col(col_name).cast("string"), fmt) for fmt in input_formats],
        F.to_date(F.col(col_name).cast("string"))
    )
    return F.when(parsed_date.isNull(), F.lit(" ")).otherwise(F.date_format(parsed_date, 'yyyy-MM-dd'))

def validate_schema(df: DataFrame, expected_schema: dict):
    """
    Validate DataFrame schema against expected_schema.
    - expected_schema is a dict: {column_name: pyspark.sql.types.DataType}
    Raises on missing/extra columns or mismatched types.
    """
    actual_fields = {field.name: field.dataType for field in df.schema.fields}
    missing = [c for c in expected_schema if c not in actual_fields]
    extra = [c for c in actual_fields if c not in expected_schema]
    type_mismatches = [c for c in expected_schema if c in actual_fields and actual_fields[c] != expected_schema[c]]

    msg = ""
    if missing:
        msg += f"Missing columns: {missing}. "
    if extra:
        msg += f"Extra columns: {extra}. "
    if type_mismatches:
        mismatches = [(c, str(actual_fields[c]), str(expected_schema[c])) for c in type_mismatches]
        msg += f"Type mismatches: {mismatches}."
    if msg:
        log_and_raise(ValueError(msg), "Schema validation failed")
    logging.info("Schema validation passed.")

def read_csv_file(path: str, spark: SparkSession) -> DataFrame:
    """
    Reads CSV using spark.read.csv, handling missing/invalid paths and logs errors.
    """
    try:
        if not path or not isinstance(path, str):
            raise FileNotFoundError("Path is not specified or invalid.")
        df = spark.read.csv(path, header=True, inferSchema=False, multiLine=True, encoding='UTF-8', ignoreLeadingWhiteSpace=True, ignoreTrailingWhiteSpace=True)
        if not df.columns or len(df.columns) < 2:
            raise ValueError(f"File at path {path} appears empty or schema is not recognized.")
        logging.info(f"Successfully read file: {path} with {len(df.columns)} columns.")
        return df
    except Exception as e:
        log_and_raise(e, f"Failed to read CSV file: {path}")

def transform_disability(df: DataFrame) -> DataFrame:
    """
    Applies all required transformations for Disability source DataFrame and returns transformed DataFrame.
    """
    try:
        # Helper Maps for special transformation rules
        benefit_map = {
            "0": "STDMaster-STD Plan 1-Core",
            "1": "STDMaster-STD Plan 1-BuyUp1",
            "2": "STDMaster-VPS Plan 1-BuyUp2",
        }
        # PACS Customer partial mapping as per rules
        pacs_map = {
            "0466964": "596028", "0503141": "0503141", "0203904": "0203904",
            "0538266": "0203904", "0407526": "0596030", "0407527": "0596030",
            "0591377": "0596030", "0472005": "0376516", "0376516": "0376516",
            "0308203": "0595985", "0308202": "0595985", "0499237": "0595985"
        }
        # Transformation
        for src, tgt, rule, tgt_type in disability_column_rules:
            if rule == "direct":
                df = df.withColumn(
                    tgt,
                    F.when((F.col(src).isNull()) | (F.col(src) == ""), F.lit(" ")).otherwise(F.col(src))
                )
            elif rule == "blank":
                df = df.withColumn(tgt, F.lit(" "))
            elif rule == "blank_float":
                df = df.withColumn(tgt, F.lit(None).cast(FloatType()))
            elif rule == "float":
                df = df.withColumn(
                    tgt,
                    F.when((F.col(src).isNull()) | (F.col(src) == ""), F.lit(None).cast(FloatType()))
                      .otherwise(F.col(src).cast("double"))
                )
            elif rule == "date":
                df = df.withColumn(
                    tgt,
                    format_date_str(src)
                )
                # For DateType columns, convert blanks to None so cast("date") works
                df = df.withColumn(
                    tgt, F.when(F.col(tgt) == " ", None).otherwise(F.col(tgt).cast(DateType()))
                )
            elif rule == "date_str":
                # Date as string conversion
                df = df.withColumn(
                    tgt,
                    format_date_str(src)
                )
            elif rule == "pad_ssn":
                # Pad SSN to 9 digits
                df = df.withColumn(
                    tgt,
                    F.when((F.col(src).isNull()) | (F.col(src) == ""), F.lit(" "))
                     .otherwise(F.lpad(F.regexp_replace(F.col(src).cast("string"), r"\D", ""), 9, "0"))
                )
            elif rule == "benefit_name_map":
                df = df.withColumn(
                    tgt,
                    F.when((F.col(src).isNull()) | (F.col(src) == ""), F.lit(" "))
                     .otherwise(
                        F.when(F.col(src) == "0", F.lit(benefit_map["0"]))
                         .when(F.col(src) == "1", F.lit(benefit_map["1"]))
                         .when(F.col(src) == "2", F.lit(benefit_map["2"]))
                         .otherwise(F.lit(" "))
                     )
                )
            elif rule == "pacs_customer_map":
                df = df.withColumn(
                    tgt,
                    F.when((F.col(src).isNull()) | (F.col(src) == ""), F.lit(" "))
                     .otherwise(
                        F.when(F.col(src).isin(list(pacs_map.keys())), F.expr(f"CASE " + ' '.join([f"WHEN `{src}` = '{k}' THEN '{v}'" for k, v in pacs_map.items()]) + " ELSE `"+src+"` END"))
                        .otherwise(F.col(src))
                     )
                )
            elif rule == "history_flag":
                df = df.withColumn(
                    tgt,
                    F.when((F.col(src).isNull()) | (F.col(src) == ""), F.lit(" "))
                     .when(F.lower(F.col(src)) == "history", F.lit("H"))
                     .when(F.lower(F.col(src)) == "takeover", F.lit("T"))
                     .otherwise(F.col(src))
                )
            else:
                df = df.withColumn(tgt, F.lit(" "))

        # Clean up dtypes and standardize nulls/blanks for all final columns
        for col_name, dtype in expected_schema.items():
            if isinstance(dtype, StringType):
                df = df.withColumn(
                    col_name,
                    F.when((F.col(col_name).isNull()) | (F.col(col_name) == ""), F.lit(" ")).otherwise(F.col(col_name).cast(StringType()))
                )
            elif isinstance(dtype, FloatType) or isinstance(dtype, DoubleType):
                df = df.withColumn(
                    col_name,
                    F.when((F.col(col_name).isNull()) | (F.trim(F.col(col_name)) == ""), F.lit(None).cast(FloatType()))
                     .otherwise(F.col(col_name).cast(FloatType()))
                )
            elif isinstance(dtype, DateType):
                # Use the logic: blanks/nulls become None
                df = df.withColumn(
                    col_name,
                    F.when((F.col(col_name).isNull()) | (F.col(col_name) == " "), F.lit(None).cast(DateType()))
                     .otherwise(F.col(col_name).cast(DateType()))
                )
            else:
                # Default: keep as string with blanks for nulls
                df = df.withColumn(
                    col_name,
                    F.when((F.col(col_name).isNull()) | (F.col(col_name) == ""), F.lit(" ")).otherwise(F.col(col_name).cast(StringType()))
                )

        # Select only the needed columns, in order; missing columns are created as blanks
        for c in column_order:
            if c not in df.columns:
                if isinstance(expected_schema[c], StringType):
                    df = df.withColumn(c, F.lit(" "))
                elif isinstance(expected_schema[c], DateType):
                    df = df.withColumn(c, F.lit(None).cast(DateType()))
                elif isinstance(expected_schema[c], FloatType):
                    df = df.withColumn(c, F.lit(None).cast(FloatType()))
                else:
                    df = df.withColumn(c, F.lit(" "))

        df = df.select(*column_order)
        return df

    except Exception as e:
        log_and_raise(e, "Error during Disability data transformation")
