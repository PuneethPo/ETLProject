def robust_sample_dataframe(df: DataFrame, n_rows: int = 10) -> DataFrame:
    # Step 1: Get one non-null row per column (if possible)
    non_null_samples = []
    
    for col_name in df.columns:
        non_null_row = df.filter(df[col_name].isNotNull()).limit(10)
        non_null_samples.append(non_null_row)
    
    # Step 2: Union all non-null examples (to ensure coverage)
    guaranteed_sample = reduce(DataFrame.unionByName, non_null_samples)

    # Step 3: Add some random samples for variety
    random_sample = df.sample(fraction=0.01).limit(n_rows)

    # Step 4: Combine and deduplicate (just in case)
    final_sample = guaranteed_sample.unionByName(random_sample).dropDuplicates()

    return final_sample.limit(n_rows)

def custom_serializer(obj):
    if isinstance(obj, (datetime, date)):
        return obj.isoformat()
    return str(obj)

def read_data(jdbc_url, table_name):
    return spark.read.format("jdbc").option("url", jdbc_url).option("dbtable", table_name).load()

def normalize_col(col):
    return col.strip().lower().replace(" ", "_")

def safe_int(val):
    try:
        if val in ("", None, "None", "nan"):
            return None
        return int(val)
    except:
        return None

def get_column_samples(df, num_samples=2):
    samples = {}
    for col in df.columns:
        key = normalize_col(col)
        samples[key] = df.select(col).dropna().limit(num_samples).rdd.flatMap(lambda x: x).collect()
    return samples

def get_column_types(df):
    return {normalize_col(f.name): str(f.dataType) for f in df.schema.fields}

def parse_LLM_response(input_string):
    # Step 1: Strip the markdown-style code block markers
    print(f"Input string: {input_string}")
    clean_json_str = input_string.strip('`').lstrip('json\n')
    # Step 2: Parse JSON
    data = json.loads(clean_json_str)
    return data

def get_profiling_metadata(table_name, project_id):
    profiling_tbl = f"Profiling_{project_id}_{table_name}" if project_id else f"Profiling_{table_name}"
    pdf = read_data(jdbc_url_target, profiling_tbl).toPandas()
    return {str(row['column_name']).strip().lower(): row for _, row in pdf.iterrows()}

def azure_search_query(search_text, top=2):
    credential = DefaultAzureCredential()
    token = credential.get_token("https://search.azure.com/.default").token
    url = f"{search_endpoint}/indexes/{search_index_name}/docs/search?api-version={search_api_version}"
    headers = {"Content-Type": "application/json", "Authorization": f"Bearer {token}"}
    payload = {"search": search_text, "top": top}
    response = requests.post(url, headers=headers, json=payload)
    if response.status_code == 200:
        return [doc.get('content', '') for doc in response.json().get('value', [])]
    else:
        logging.warning(f"Azure Search Error {response.status_code}: {response.text}")
        return []

def get_rag_contexts(target_columns):
    return {tgt_col: azure_search_query(tgt_col) for tgt_col in target_columns}

def extract_json_block(response_text):
    try:
        block = re.search(r"```json(.*?)```", response_text, re.DOTALL)
        return json.loads(block.group(1).strip()) if block else json.loads(response_text.strip())
    except:
        return []
