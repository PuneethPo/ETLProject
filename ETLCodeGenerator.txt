import nbformat
from nbformat.v4 import new_notebook, new_code_cell

# Read the .py file
with open(f"{to_be_generated_python_file_name}.py", "r") as f:
    lines = f.readlines()

cells = []
current_cell_lines = []

for line in lines:
    if line.strip().startswith(generated_code_blocks_seperator):
        # Save the current cell if it has content
        if current_cell_lines:
            cells.append(new_code_cell("".join(current_cell_lines)))
            current_cell_lines = []
    current_cell_lines.append(line)

# Add the last cell if any lines remain
if current_cell_lines:
    cells.append(new_code_cell("".join(current_cell_lines)))

# Create a new notebook
notebook = new_notebook(cells=cells)

# Save as .ipynb
with open(f"{to_be_generated_ipynb_file_name}.ipynb", "w") as f:
    nbformat.write(notebook, f)
