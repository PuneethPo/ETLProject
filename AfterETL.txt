def format_date_str(col_name, input_formats=None):
    """
    Accepts a column (string or date) and returns formatted YYYY-MM-DD string (not timestamp!).
    If value is null or unparsable, returns blank.
    """
    input_formats = input_formats or ["M/d/yyyy", "MM/dd/yyyy", "yyyy-MM-dd", "MM-dd-yyyy"]
    parsed_date = F.coalesce(
        *[F.to_date(F.col(col_name).cast("string"), fmt) for fmt in input_formats],
        F.col(col_name).cast("date")
    )
    return F.when(
        parsed_date.isNull(),
        F.lit(" ")
    ).otherwise(
        F.date_format(parsed_date, 'yyyy-MM-dd')
    )

def validate_schema(df: DataFrame, expected_schema: dict):
    """
    Validates DataFrame schema against expected column names and types.
    Raises informative errors if mismatches detected.
    """
    df_fields = {field.name: field.dataType for field in df.schema.fields}
    
    expected_cols = set(expected_schema.keys())
    actual_cols = set(df_fields.keys())

    extra_cols = actual_cols - expected_cols
    missing_cols = expected_cols - actual_cols

    if missing_cols:
        raise ValueError(f"Missing columns in DataFrame: {missing_cols}")
    if extra_cols:
        raise ValueError(f"Extra columns present in DataFrame: {extra_cols}")

    mismatch_types = []
    for col, spark_type in expected_schema.items():
        if not isinstance(df_fields[col], spark_type):
            mismatch_types.append(f"{col}: expected {spark_type}, found {type(df_fields[col])}")
    if mismatch_types:
        raise TypeError("Schema type mismatches detected: " + "; ".join(mismatch_types))
    logging.info('Schema validation passed.')
