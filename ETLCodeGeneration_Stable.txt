def upload_blob(container_client, blob_name, content):
    try:
        blob_client = container_client.get_blob_client(blob_name)
        blob_client.upload_blob(content, overwrite=True)
        logging.info(f"Uploaded blob: {blob_name}")
    except Exception as e:
        logging.error(f"Error uploading blob: {e}")

def list_blobs(container_client):
    try:
        blobs = container_client.list_blobs()
        blob_names = [blob.name for blob in blobs]
        logging.info(f"Blobs in container: {blob_names}")
        return blob_names
    except Exception as e:
        logging.error(f"Error listing blobs: {e}")
        return []

def download_blob(container_client, blob_name):
    try:
        blob_client = container_client.get_blob_client(blob_name)
        content = blob_client.download_blob().readall().decode('utf-8')
        logging.info(f"Downloaded blob content: {content}")
        return content
    except Exception as e:
        logging.error(f"Error downloading blob: {e}")
        return None
